Un solo punto

Mapa de un solo punto sin sin el uso de popUp o slider. 

<!-- SCRIPTS -->
<script>
    const latitude = -34.60759;
    const longitude = -58.37393;
    const mapOptions = {
      "scope": "poncho-map",
      "no_info": true,
      "map_zoom": 17, 
      "map_view": [latitude, longitude]
    };
    const filter = new PonchoMap([{"latitud": latitude, "longitud": longitude}], mapOptions);
    filter.render();
</script>
<!-- / SCRIPTS -->
